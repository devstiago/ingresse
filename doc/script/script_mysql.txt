CREATE TABLE TABEVENTO (
    CODIGO INT AUTO_INCREMENT PRIMARY KEY,
    NOMEEVENTO VARCHAR(100),
    DESCRICAO VARCHAR(2000),
    DATA VARCHAR(30),
    HORARIO VARCHAR(30),
    LOCAL VARCHAR(100),
    IMGTOPO VARCHAR(100)
);

CREATE TABLE TABCONFIG_TICKET (
    CODIGO INT AUTO_INCREMENT PRIMARY KEY,
    NOMETICKET VARCHAR(100),
    DESCRICAO VARCHAR(100),
    VL_TAXASERV DECIMAL(10, 2),
    VL_INGRESSO DECIMAL(10, 2),
    VL_DESCONTO DECIMAL(10, 2),
    VL_TOTAL DECIMAL(10, 2)
);

CREATE TABLE TABPAGAMENTO (
    CODIGO INT AUTO_INCREMENT PRIMARY KEY,
    STSPAGAMENTO INT,
    VALOR DECIMAL(10, 2),
    CODIGOB64 VARCHAR(2000),
    CODIGOPIX VARCHAR(2000),
    DATAHORA VARCHAR(30)
);

CREATE TABLE TABINGRESSOS (
    CODIGO INT AUTO_INCREMENT PRIMARY KEY,
    NOME VARCHAR(100),
    CPF VARCHAR(30),
    WHATTSAP VARCHAR(30),
    EMAIL VARCHAR(100),
    CHECK_IN VARCHAR(30),
    STSINGRESSO CHAR(1),
    OBS VARCHAR(200)
);
